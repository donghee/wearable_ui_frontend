# SBCL 설치

다운로드 URL : https://www.sbcl.org/platform-table.html

우분투 : sudo apt install sbcl

# lisp 다운로드 및 패키지 설치

curl -O https://beta.quicklisp.org/quicklisp.lisp

sbcl --load quicklisp.lisp

# 설치 중 문구

(quicklisp-quickstart:install :path "/home/zento/test/quicklisp/")

(ql:add-to-init-file)

# 패키지 설치

sbcl --quit --load quicklisp/setup.lisp --eval "(ql:quickload '(:ningle :djula :dexador :cl-json))"
